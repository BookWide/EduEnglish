第一次使用：Supabase 建表 SQL
create table if not exists public.community_groups (
  id text primary key,
  name text not null,
  status text not null default 'active',
  created_by text,
  created_at timestamptz not null default now()
);

create table if not exists public.community_group_members (
  group_id text not null references public.community_groups(id) on delete cascade,
  user_email text not null,
  role text not null default 'member' check (role in ('member','group_admin')),
  created_at timestamptz not null default now(),
  primary key (group_id,user_email)
);

alter table public.community_groups enable row level security;
alter table public.community_group_members enable row level security;

create policy "admins manage community groups" on public.community_groups
for all to authenticated
using (exists (select 1 from public.profiles p where lower(p.email)=lower(auth.jwt()->>'email') and p.is_admin=true and coalesce(p.is_paused,false)=false))
with check (exists (select 1 from public.profiles p where lower(p.email)=lower(auth.jwt()->>'email') and p.is_admin=true and coalesce(p.is_paused,false)=false));

create policy "admins manage community members" on public.community_group_members
for all to authenticated
using (exists (select 1 from public.profiles p where lower(p.email)=lower(auth.jwt()->>'email') and p.is_admin=true and coalesce(p.is_paused,false)=false))
with check (exists (select 1 from public.profiles p where lower(p.email)=lower(auth.jwt()->>'email') and p.is_admin=true and coalesce(p.is_paused,false)=false));
